function regenData() {
  const mainA = attrs[Math.floor(Math.random()*attrs.length)];
  const mainV = mainA.values[0].val;
  rows = [];
  for(let i=0; i<8; i++) {
    const r = {};
    attrs.forEach(a => r[a.key] = a.values[Math.floor(Math.random()*2)].val);
    r.ergebnis = (r[mainA.key] === mainV) ? (Math.random()<0.2?'nein':'ja') : (Math.random()<0.2?'ja':'nein');
    rows.push(r);
  }
  renderTable();
  rebuildAll();
}

// GINI ALGORITHMUS
function gini(data) {
  if (data.length === 0) return 0;
  const pJa = data.filter(d => d.ergebnis === 'ja').length / data.length;
  return 1 - Math.pow(pJa, 2) - Math.pow(1 - pJa, 2);
}
function buildTree(data, usedAttrs, depth) {
  const gVal = gini(data);
  if (gVal === 0 || depth >= attrs.length) {
    return { type: 'leaf', label: data.filter(d=>d.ergebnis==='ja').length >= data.length/2 ? 'ja' : 'nein', gv: gVal, n: data.length };
  }
  let best = null, minGini = Infinity;
  attrs.forEach(a => {
    if (usedAttrs.has(a.key)) return;
    const v0 = a.values[0].val;
    const leftD = data.filter(d => d[a.key] === v0);
    const rightD = data.filter(d => d[a.key] !== v0);
    if (leftD.length === 0 || rightD.length === 0) return;
    const wg = (leftD.length/data.length)*gini(leftD) + (rightD.length/data.length)*gini(rightD);
    if (wg < minGini) { minGini = wg; best = { attr: a, leftD, rightD }; }
  });
  if (!best) return { type: 'leaf', label: data.filter(d=>d.ergebnis==='ja').length >= data.length/2 ? 'ja' : 'nein', gv: gVal, n: data.length };
  const nextUsed = new Set(usedAttrs); nextUsed.add(best.attr.key);
  return {
    type: 'node', attr: best.attr, gv: gVal, n: data.length,
    left: buildTree(best.leftD, nextUsed, depth + 1),
    right: buildTree(best.rightD, nextUsed, depth + 1)
  };
}
// STUDENT TREE UI
function createStudentStructure(depth, level) {
  const id = 'n' + (nc++);
  if (level >= attrs.length) return { type: 'leaf', id, level };
  return { type: 'node', id, level, merkmal: null, left: createStudentStructure(depth, level+1), right: createStudentStructure(depth, level+1) };
}
function renderStudentTree() {
  nc = 0; snodeMap = {};
  const depth = attrs.length;
  stree = createStudentStructure(depth, 0);
  
  const cw = Math.max(800, Math.pow(2, depth) * 160);
  const ch = (depth + 1) * ROW + PAD;
  const canvas = document.getElementById('student-canvas');
  const svg = document.getElementById('student-svg');
  
  canvas.style.width = cw + 'px'; canvas.style.height = ch + 'px';
  svg.setAttribute('width', cw); svg.setAttribute('height', ch);
  canvas.querySelectorAll('.tn').forEach(n => n.remove());
  svg.innerHTML = '';
  assignPositions(stree, cw/2, PAD, cw/4);
  drawStudentNodes(stree, null, canvas, svg);
}
function assignPositions(node, x, y, offset) {
  node._x = x; node._y = y;
  if (node.left) {
    assignPositions(node.left, x - offset, y + ROW, offset/2);
    assignPositions(node.right, x + offset, y + ROW, offset/2);
  }
}
function drawStudentNodes(node, parent, canvas, svg) {
  if (parent) {
    let label = "?";
    if (parent.merkmal) {
      const a = attrs.find(at => at.key === parent.merkmal);
      label = (node === parent.left) ? a.values[0].label : a.values[1].label;
    }
    drawLine(svg, parent._x, parent._y + 60, node._x, node._y, label);
  }
  const div = document.createElement('div');
  div.className = 'tn' + (node.level === 0 ? ' tn-root' : '');
  div.id = node.id;
  div.style.left = (node._x - NW/2) + 'px'; div.style.top = node._y + 'px';
  
  if (node.type === 'node') {
    div.innerHTML = `<span class="tn-lbl">Attribut wählen</span>
      <select onchange="updateStudentNode('${node.id}', this.value)">
        <option value="">-- ? --</option>
        ${attrs.map(a => `<option value="${a.key}">${a.label}</option>`).join('')}

  } else {
    div.innerHTML = `<span class="tn-lbl">Klasse</span>

        <option value="">-- ? --</option>
        <option value="ja">Happy</option>
        <option value="nein">Not Happy</option>
      </select>`;
  }
  snodeMap[node.id] = node;
  canvas.appendChild(div);
  if (node.left) { drawStudentNodes(node.left, node, canvas, svg); drawStudentNodes(node.right, node, canvas, svg); }
}
function updateStudentNode(id, val) {
  snodeMap[id].merkmal = val;
  const svg = document.getElementById('student-svg');
  svg.innerHTML = '';
  refreshLines(stree, null, svg);
}
function updateStudentLeaf(id, val) { snodeMap[id].answer = val; }
function refreshLines(node, parent, svg) {
  if (parent) {
    let label = "?";
    if (parent.merkmal) {
      const a = attrs.find(at => at.key === parent.merkmal);
      label = (node === parent.left) ? a.values[0].label : a.values[1].label;
    }
    drawLine(svg, parent._x, parent._y + 60, node._x, node._y, label);
  }
  if (node.left) { refreshLines(node.left, node, svg); refreshLines(node.right, node, svg); }
}
function drawLine(svg, x1, y1, x2, y2, txt) {
  const ns = 'http://www.w3.org/2000/svg';
  const l = document.createElementNS(ns, 'line');
  l.setAttribute('x1', x1); l.setAttribute('y1', y1); l.setAttribute('x2', x2); l.setAttribute('y2', y2);
  l.setAttribute('stroke', 'rgba(139,92,246,0.3)'); l.setAttribute('stroke-width', '2');
  svg.appendChild(l);
  const t = document.createElementNS(ns, 'text');
  t.setAttribute('x', (x1+x2)/2 + 10); t.setAttribute('y', (y1+y2)/2);
  t.setAttribute('fill', 'var(--ac2)'); t.setAttribute('font-size', '11');

  svg.appendChild(t);

function checkTree() {
  let isCorrect = true;
  function validate(s, sol) {
    const el = document.getElementById(s.id);
    el.classList.remove('tn-ok', 'tn-fail');
    if (s.type === 'leaf') {
      const ok = s.answer === sol.label;
      el.classList.add(ok ? 'tn-ok' : 'tn-fail');
      if (!ok) isCorrect = false;

      const ok = sol.type === 'node' && s.merkmal === sol.attr.key;
      el.classList.add(ok ? 'tn-ok' : 'tn-fail');
      if (!ok) isCorrect = false;
      validate(s.left, sol.left || sol); validate(s.right, sol.right || sol);
    }

  rb.style.display = 'block';

}
